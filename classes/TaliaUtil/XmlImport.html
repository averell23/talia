30:             progress.inc
31:             begin
32:               TaliaUtil::HyperImporter::Importer.import(REXML::Document.new(source.elements[1].to_s), options)
33:             rescue Exception => e
34:               $stderr.puts("Error when importing from #{xml_file}: #{e}\nBacktrace: #{e.backtrace.join("\n")}")
35:             end
36:           end

38:         end
