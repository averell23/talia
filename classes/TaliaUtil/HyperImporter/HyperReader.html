 98:       def add_hyper_file
 99:         url = from_element('file_url')
100:         return unless(url) # No file found
101:         options = {}
102:         location = from_element('file_name')
103:         options[:location] = location if(location)
104:         # Try to get the content type from the data
105:         content_type = from_element('file_content_type')
106:         content_type.downcase! if(content_type)
107:         mime_type = case content_type
108:         when 'image', nil
109:           ext = location ? File.extname(location)[1..-1] : File.extname(url)[1..-1]
110:           Mime::Type.lookup_by_extension(ext)
111:         when 'generic xml'
112:           Mime::Type.lookup('text/xml')
113:         else
114:           Mime::Type.lookup_by_extension(content_type)
115:         end
116:         options[:mime_type] = mime_type
117:         add_file url, options
118:         add N::DCNS.format, mime_type.to_s
